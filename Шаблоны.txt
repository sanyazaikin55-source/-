(function () {
  const T = (window.externalTemplates = window.externalTemplates || {});

  /* ---------- helpers (если в проекте уже есть — не переопределяем) ---------- */
  if (typeof escHtml !== "function") {
    window.escHtml = function (s) {
      return String(s ?? "").replace(/[&<>"']/g, (c) => ({ "&": "&amp;", "<": "&lt;", ">": "&gt;", '"': "&quot;", "'": "&#39;" }[c]));
    };
  }
  if (typeof val !== "function") {
    window.val = function (data, key, ph = "_______________") {
      const v = (data?.[key] ?? "").toString().trim();
      return v ? escHtml(v) : ph;
    };
  }
  if (typeof money !== "function") {
    window.money = function (data, key, ph = "_______") {
      const v = (data?.[key] ?? "").toString().trim();
      return v ? escHtml(v) : ph;
    };
  }
  if (typeof dateRu !== "function") {
    window.dateRu = function (v) {
      const s = (v ?? "").toString().trim();
      if (/^\d{4}-\d{2}-\d{2}$/.test(s)) {
        const [y, m, d] = s.split("-");
        return `${d}.${m}.${y}`;
      }
      return s || "_______________";
    };
  }
  if (typeof rightBlock !== "function") {
    window.rightBlock = function (lines) {
      return (lines || []).map((t) => `<div class="party-line">${t}</div>`).join("");
    };
  }
  if (typeof p !== "function") {
    window.p = function (t) {
      return `<div class="text-block">${t}</div>`;
    };
  }
  if (typeof reqBlock !== "function") {
    window.reqBlock = function (items) {
      return (items || []).map((t, i) => `<div class="request-item">${i + 1}. ${t}</div>`).join("");
    };
  }
  if (typeof signBlock !== "function") {
    window.signBlock = function (leftText, rightText) {

        </div>

    };
  }

  /* ---------- token renderer: {{field}}, {{date:field}}, {{money:field}} ---------- */
  function tkn(str, data) {
    return String(str || "")
      .replace(/\{\{date:([a-zA-Z0-9_]+)\}\}/g, (_, k) => dateRu(data?.[k]))
      .replace(/\{\{money:([a-zA-Z0-9_]+)\}\}/g, (_, k) => money(data, k))
      .replace(/\{\{([a-zA-Z0-9_]+)\}\}/g, (_, k) => val(data, k));

  function renderDoc(def, data) {
    const right = (def.right || []).map((s) => tkn(s, data));
    const paras = (def.paragraphs || []).map((s) => p(tkn(s, data))).join("");
    const legal = def.legal ? `<div class="legal-ref">${tkn(def.legal, data)}</div>` : "";
    const requestTitle = def.requestTitle ? `<div class="request-title">${def.requestTitle}</div>` : `<div class="request-title">ПРОШУ:</div>`;
    const req = def.requests ? reqBlock((def.requests || []).map((s) => tkn(s, data))) : "";
    const signL = tkn(def.signLeft || `Дата: ${new Date().toLocaleDateString("ru-RU")}`, data);
    const signR = tkn(def.signRight || `Подпись: ___________`, data);

    return `
      ${right.length ? rightBlock(right) : ""}
      <div class="doc-title">${tkn(def.docTitle || def.title || "ДОКУМЕНТ", data)}</div>
      ${paras}
      ${legal}
      ${req ? requestTitle + req : ""}
      ${signBlock(signL, signR)}
    `;
  }
